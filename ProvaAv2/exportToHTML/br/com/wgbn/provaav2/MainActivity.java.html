62       private void showLogin(){ 
63           FragmentTransaction ft = getSupportFragmentManager().beginTransaction(); 
64           Fragment prev = getSupportFragmentManager().findFragmentByTag("dialog"); 
65           if (prev != null) { 
66               ft.remove(prev); 
67           } 
68           ft.addToBackStack(null); 
69    
70           // Create and show the dialog. 
71           DialogFragment newFragment = new LoginDialogFragment(); 
72           newFragment.setCancelable(false); 
73           newFragment.show(ft, "dialog"); 
74       }

111      public boolean onNavigationItemSelected(MenuItem item) { 
112          // Handle navigation view item clicks here. 
113          int id = item.getItemId(); 
114   
115          if (id == R.id.btnLista) { 
116              Fragment listar = ListarFragment.getInstance(); 
117              fragmentManager.beginTransaction().replace(R.id.contents, listar).commit(); 
118          } else if (id == R.id.btnCad) { 
119              Fragment cadastrar = new CadastrarFragment(); //CadastrarFragment.getInstance(); 
120              fragmentManager.beginTransaction().replace(R.id.contents, cadastrar).commit(); 
121          } 
122   
123          DrawerLayout drawer = (DrawerLayout) findViewById(R.id.drawer_layout); 
124          drawer.closeDrawer(GravityCompat.START); 
125          return true;
