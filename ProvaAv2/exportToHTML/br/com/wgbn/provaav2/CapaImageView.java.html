1    package br.com.wgbn.provaav2; 
2     
3    import android.content.Context; 
4    import android.graphics.drawable.Drawable; 
5    import android.util.AttributeSet; 
6    import android.view.View; 
7     
8    import com.makeramen.roundedimageview.RoundedImageView; 
9     
10   /** 
11    * Classe que extende do componente RoundedImageView para redimencionar 
12    * automaticamente a imagem para caber dentro do CardView. 
13    * Essa classe foi retirada do Stackoverflow: 
14    * http://stackoverflow.com/questions/13992535/android-imageview-scale-smaller-image-to-width-with-flexible-height-without-crop 
15    */ 
16   public class CapaImageView extends RoundedImageView { 
17    
18       public CapaImageView(final Context context, final AttributeSet attrs) { 
19           super(context, attrs); 
20       } 
21    
22       @Override 
23       protected void onMeasure(final int widthMeasureSpec, final int heightMeasureSpec) { 
24           final Drawable d = this.getDrawable(); 
25    
26           if (d != null) { 
27               // ceil not round - avoid thin vertical gaps along the left/right edges 
28               final int width = View.MeasureSpec.getSize(widthMeasureSpec); 
29               final int height = (int) Math.ceil(width * (float) d.getIntrinsicHeight() / d.getIntrinsicWidth()); 
30               this.setMeasuredDimension(width, height); 
31           } else { 
32               super.onMeasure(widthMeasureSpec, heightMeasureSpec); 
33           } 
34       } 
35   }
