42       @Override 
43       public View onCreateView(LayoutInflater inflater, final ViewGroup container, Bundle savedInstanceState) { 
44           onViewStateRestored(savedInstanceState); 
45           View vi = inflater.inflate(R.layout.layout_fragment_cadastrar, container, false); 
46    
47           btnCadSalvar = (Button)vi.findViewById(R.id.btnCadSalvar); 
48           btnCadCancelar = (Button)vi.findViewById(R.id.btnCadCancelar); 
49           btnCadBuscar = (Button)vi.findViewById(R.id.btnCadBuscar); 
50    
51           txtTitulo = (EditText)vi.findViewById(R.id.txtTitulo); 
52           txtSubTitulo = (EditText)vi.findViewById(R.id.txtSubTitulo); 
53           txtEditora = (EditText)vi.findViewById(R.id.txtEditora); 
54           txtCapa = (EditText)vi.findViewById(R.id.txtCapa); 
55           txtAno = (EditText)vi.findViewById(R.id.txtAno); 
56           txtNumero = (EditText)vi.findViewById(R.id.txtNumero); 
57    
58           btnCadSalvar.setOnClickListener(new View.OnClickListener() { 
59               @Override 
60               public void onClick(View v) { 
61                   if (validaCampos()){ 
62                       EdicaoModel e = new EdicaoModel(); 
63                       e.setTitulo(txtTitulo.getText().toString()); 
64                       e.setSubtitulo(txtSubTitulo.getText().toString()); 
65                       e.setEditora(txtEditora.getText().toString()); 
66                       e.setCapa(txtCapa.getText().toString()); 
67                       e.setAno(new Integer(txtAno.getText().toString()).intValue()); 
68                       e.setNumero(new Integer(txtNumero.getText().toString()).intValue()); 
69                       EdicaoService.getInstance().add(e); 
70                       voltarListagem(v); 
71                   } else { 
72                       Toast.makeText(getContext(), "Você deve preencher todos os campos!", Toast.LENGTH_LONG).show(); 
73                   } 
74               } 
75           }); 
76    
77           btnCadCancelar.setOnClickListener(new View.OnClickListener() { 
78               @Override 
79               public void onClick(View v) { 
80                   voltarListagem(v); 
81               } 
82           }); 
83    
84           btnCadBuscar.setOnClickListener(new View.OnClickListener() { 
85               @Override 
86               public void onClick(View v) { 
87                   if (!txtTitulo.getText().toString().isEmpty() && !txtSubTitulo.getText().toString().isEmpty()){ 
88                       String url = "https://www.google.com.br/search?q="+txtTitulo.getText().toString()+" "+txtSubTitulo.getText().toString()+" cover capa&biw=300&bih=785&source=lnms&tbm=isch&sa=X"; 
89                       Intent i = new Intent(Intent.ACTION_VIEW); 
90                       i.setData(Uri.parse(url)); 
91                       startActivity(i); 
92                   } 
93               } 
94           }); 
95    
96           return vi; 
97       } 
98    
99       private void voltarListagem(View v){ 
100          FragmentManager fragmentManager = getFragmentManager(); 
101   
102          Fragment listar = ListarFragment.getInstance(); 
103          fragmentManager.beginTransaction().replace(R.id.contents, listar).commit(); 
104      } 
105   
106      private boolean validaCampos(){ 
107          return !( 
108              txtTitulo.getText().toString().isEmpty() || 
109              txtAno.getText().toString().isEmpty() || 
110              txtNumero.getText().toString().isEmpty() || 
111              txtCapa.getText().toString().isEmpty() || 
112              txtEditora.getText().toString().isEmpty() || 
113              txtSubTitulo.getText().toString().isEmpty() 
114          ); 
115      }
