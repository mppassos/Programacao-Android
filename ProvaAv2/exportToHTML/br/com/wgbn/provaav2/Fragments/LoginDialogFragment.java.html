1    package br.com.wgbn.provaav2.Fragments; 
2     
3    import android.app.Dialog; 
4    import android.content.SharedPreferences; 
5    import android.os.Bundle; 
6    import android.support.annotation.NonNull; 
7    import android.support.v4.app.DialogFragment; 
8    import android.support.v7.app.AlertDialog; 
9    import android.view.LayoutInflater; 
10   import android.view.View; 
11   import android.widget.Button; 
12   import android.widget.CheckBox; 
13   import android.widget.EditText; 
14   import android.widget.Toast; 
15    
16   import br.com.wgbn.provaav2.R; 
17    
18   public class LoginDialogFragment extends DialogFragment { 
19    
20       private SharedPreferences shared; 
21       private EditText usuario; 
22       private EditText senha; 
23       private CheckBox chk; 
24    
25       @NonNull 
26       @Override 
27       public Dialog onCreateDialog(Bundle savedInstanceState) { 
28           LayoutInflater inflater = getActivity().getLayoutInflater(); 
29           View view = inflater.inflate(R.layout.dialogo_login, null); 
30    
31           usuario = (EditText)view.findViewById(R.id.txtUsuario); 
32           senha = (EditText)view.findViewById(R.id.txtSenha); 
33           Button btnEntrar = (Button)view.findViewById(R.id.btnLoginEntrar); 
34           chk = (CheckBox)view.findViewById(R.id.chkLembrar); 
35    
36           if (verificaSalvo()){ 
37               usuario.setText(shared.getString("usuario", "")); 
38               senha.setText(shared.getString("senha", "")); 
39               chk.setChecked(true); 
40           } 
41    
42           btnEntrar.setOnClickListener(new View.OnClickListener() { 
43               @Override 
44               public void onClick(View v) { 
45                   if (!usuario.getText().toString().isEmpty() && !senha.getText().toString().isEmpty()){ 
46                       if (chk.isChecked()){ 
47                           SharedPreferences.Editor editor = shared.edit(); 
48                           editor.putString("usuario", usuario.getText().toString()); 
49                           editor.putString("senha", senha.getText().toString()); 
50                           editor.commit(); 
51                       } 
52    
53                       if (usuario.getText().toString().equals("senai") && senha.getText().toString().equals("senai")){ 
54                           fechar(); 
55                       } else { 
56                           Toast.makeText(getContext(), "Usuário ou senha inválidos!", Toast.LENGTH_LONG).show(); 
57                       } 
58                   } else { 
59                       Toast.makeText(getContext(), "Preencha todos os campos!", Toast.LENGTH_LONG).show(); 
60                   } 
61               } 
62           }); 
63    
64           chk.setOnClickListener(new View.OnClickListener() { 
65               @Override 
66               public void onClick(View v) { 
67                   if (!chk.isChecked() && verificaSalvo()){ 
68                       SharedPreferences.Editor editor = shared.edit(); 
69                       editor.remove("usuario"); 
70                       editor.remove("senha"); 
71                       editor.commit(); 
72                   } 
73               } 
74           }); 
75    
76           AlertDialog.Builder builder = new AlertDialog.Builder(getContext()); 
77           builder.setView(view); 
78    
79           return builder.create(); 
80       } 
81    
82       private boolean verificaSalvo(){ 
83           shared = getContext().getSharedPreferences("av2", 0); 
84           return shared.getString("usuario", null) != null; 
85       } 
86    
87       private void fechar(){ 
88           this.dismiss(); 
89       } 
90   }
